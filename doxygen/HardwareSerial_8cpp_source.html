00066 inline void store_char(unsigned char c, ring_buffer *rx_buffer)
00067 {
00068   int i = (unsigned int)(rx_buffer->head + 1) % RX_BUFFER_SIZE;
00069 
00070   // if we should be storing the received character into the location
00071   // just before the tail (meaning that the head would advance to the
00072   // current location of the tail), we're about to overflow the buffer
00073   // and so we don't write the character or advance the head.
00074   if (i != rx_buffer->tail) {
00075     rx_buffer->buffer[rx_buffer->head] = c;
00076     rx_buffer->head = i;
00077   }
00078 }

00194 void HardwareSerial::begin(long baud)
00195 {
00196   uint16_t baud_setting;
00197   bool use_u2x = true;
00198 
00199 #if F_CPU == 16000000UL
00200   // hardcoded exception for compatibility with the bootloader shipped
00201   // with the Duemilanove and previous boards and the firmware on the 8U2
00202   // on the Uno and Mega 2560.
00203   if (baud == 57600) {
00204     use_u2x = false;
00205   }
00206 #endif
00207   
00208   if (use_u2x) {
00209     *_ucsra = 1 << _u2x;
00210     baud_setting = (F_CPU / 4 / baud - 1) / 2;
00211   } else {
00212     *_ucsra = 0;
00213     baud_setting = (F_CPU / 8 / baud - 1) / 2;
00214   }
00215 
00216   // assign the baud_setting, a.k.a. ubbr (USART Baud Rate Register)
00217   *_ubrrh = baud_setting >> 8;
00218   *_ubrrl = baud_setting;
00219 
00220   sbi(*_ucsrb, _rxen);
00221   sbi(*_ucsrb, _txen);
00222   sbi(*_ucsrb, _rxcie);
00223 }

00246 int HardwareSerial::read(void)
00247 {
00248   // if the head isn't ahead of the tail, we don't have any characters
00249   if (_rx_buffer->head == _rx_buffer->tail) {
00250     return -1;
00251   } else {
00252     unsigned char c = _rx_buffer->buffer[_rx_buffer->tail];
00253     _rx_buffer->tail = (unsigned int)(_rx_buffer->tail + 1) % RX_BUFFER_SIZE;
00254     return c;
00255   }
00256 }

00258 void HardwareSerial::flush()
00259 {
00260   // don't reverse this or there may be problems if the RX interrupt
00261   // occurs after reading the value of rx_buffer_head but before writing
00262   // the value to rx_buffer_tail; the previous value of rx_buffer_head
00263   // may be written to rx_buffer_tail, making it appear as if the buffer
00264   // don't reverse this or there may be problems if the RX interrupt
00265   // occurs after reading the value of rx_buffer_head but before writing
00266   // the value to rx_buffer_tail; the previous value of rx_buffer_head
00267   // may be written to rx_buffer_tail, making it appear as if the buffer
00268   // were full, not empty.
00269   _rx_buffer->head = _rx_buffer->tail;
00270 }
