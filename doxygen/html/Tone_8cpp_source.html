00115 static int8_t toneBegin(uint8_t _pin)
00116 {
00117   int8_t _timer = -1;
00118 
00119   // if we're already using the pin, the timer should be configured.  
00120   for (int i = 0; i < AVAILABLE_TONE_PINS; i++) {
00121     if (tone_pins[i] == _pin) {
00122       return pgm_read_byte(tone_pin_to_timer_PGM + i);
00123     }
00124   }
00125   
00126   // search for an unused timer.
00127   for (int i = 0; i < AVAILABLE_TONE_PINS; i++) {
00128     if (tone_pins[i] == 255) {
00129       tone_pins[i] = _pin;
00130       _timer = pgm_read_byte(tone_pin_to_timer_PGM + i);
00131       break;
00132     }
00133   }
00134   
00135   if (_timer != -1)
00136   {
00137     // Set timer specific stuff
00138     // All timers in CTC mode
00139     // 8 bit timers will require changing prescalar values,
00140     // whereas 16 bit timers are set to either ck/1 or ck/64 prescalar
00141     switch (_timer)
00142     {
00143       #if defined(TCCR0A) && defined(TCCR0B)
00144       case 0:
00145         // 8 bit timer
00146         TCCR0A = 0;
00147         TCCR0B = 0;
00148         bitWrite(TCCR0A, WGM01, 1);
00149         bitWrite(TCCR0B, CS00, 1);
00150         timer0_pin_port = portOutputRegister(digitalPinToPort(_pin));
00151         timer0_pin_mask = digitalPinToBitMask(_pin);
00152         break;
00153       #endif
00154 
00155       #if defined(TCCR1A) && defined(TCCR1B) && defined(WGM12)
00156       case 1:
00157         // 16 bit timer
00158         TCCR1A = 0;
00159         TCCR1B = 0;
00160         bitWrite(TCCR1B, WGM12, 1);
00161         bitWrite(TCCR1B, CS10, 1);
00162         timer1_pin_port = portOutputRegister(digitalPinToPort(_pin));
00163         timer1_pin_mask = digitalPinToBitMask(_pin);
00164         break;
00165       #endif
00166 
00167       #if defined(TCCR2A) && defined(TCCR2B)
00168       case 2:
00169         // 8 bit timer
00170         TCCR2A = 0;
00171         TCCR2B = 0;
00172         bitWrite(TCCR2A, WGM21, 1);
00173         bitWrite(TCCR2B, CS20, 1);
00174         timer2_pin_port = portOutputRegister(digitalPinToPort(_pin));
00175         timer2_pin_mask = digitalPinToBitMask(_pin);
00176         break;
00177       #endif
00178 
00179       #if defined(TCCR3A) && defined(TCCR3B) &&  defined(TIMSK3)
00180       case 3:
00181         // 16 bit timer
00182         TCCR3A = 0;
00183         TCCR3B = 0;
00184         bitWrite(TCCR3B, WGM32, 1);
00185         bitWrite(TCCR3B, CS30, 1);
00186         timer3_pin_port = portOutputRegister(digitalPinToPort(_pin));
00187         timer3_pin_mask = digitalPinToBitMask(_pin);
00188         break;
00189       #endif
00190 
00191       #if defined(TCCR4A) && defined(TCCR4B) &&  defined(TIMSK4)
00192       case 4:
00193         // 16 bit timer
00194         TCCR4A = 0;
00195         TCCR4B = 0;
00196         #if defined(WGM42)
00197           bitWrite(TCCR4B, WGM42, 1);
00198         #elif defined(CS43)
00199           #warning this may not be correct
00200           // atmega32u4
00201           bitWrite(TCCR4B, CS43, 1);
00202         #endif
00203         bitWrite(TCCR4B, CS40, 1);
00204         timer4_pin_port = portOutputRegister(digitalPinToPort(_pin));

00207       #endif
00208 
00209       #if defined(TCCR5A) && defined(TCCR5B) &&  defined(TIMSK5)
00210       case 5:
00211         // 16 bit timer

00230 void tone(uint8_t _pin, unsigned int frequency, unsigned long duration)
00231 {
00232   uint8_t prescalarbits = 0b001;
00233   long toggle_count = 0;
00234   uint32_t ocr = 0;
00235   int8_t _timer;
00236 
00237   _timer = toneBegin(_pin);
00238 
00239   if (_timer >= 0)
00240   {
00241     // Set the pinMode as OUTPUT
00242     pinMode(_pin, OUTPUT);
00243     
00244     // if we are using an 8 bit timer, scan through prescalars to find the best fit
00245     if (_timer == 0 || _timer == 2)
00246     {
00247       ocr = F_CPU / frequency / 2 - 1;
00248       prescalarbits = 0b001;  // ck/1: same for both timers
00249       if (ocr > 255)
00250       {
00251         ocr = F_CPU / frequency / 2 / 8 - 1;
00252         prescalarbits = 0b010;  // ck/8: same for both timers
00253 
00254         if (_timer == 2 && ocr > 255)
00255         {
00256           ocr = F_CPU / frequency / 2 / 32 - 1;
00257           prescalarbits = 0b011;
00258         }
00259 
00260         if (ocr > 255)
00261         {
00262           ocr = F_CPU / frequency / 2 / 64 - 1;
00263           prescalarbits = _timer == 0 ? 0b011 : 0b100;
00264 
00265           if (_timer == 2 && ocr > 255)
00266           {
00267             ocr = F_CPU / frequency / 2 / 128 - 1;
00268             prescalarbits = 0b101;
00269           }
00270 
00271           if (ocr > 255)
00272           {
00273             ocr = F_CPU / frequency / 2 / 256 - 1;
00274             prescalarbits = _timer == 0 ? 0b100 : 0b110;
00275             if (ocr > 255)
00276             {
00277               // can't do any better than /1024
00278               ocr = F_CPU / frequency / 2 / 1024 - 1;
00279               prescalarbits = _timer == 0 ? 0b101 : 0b111;
00280             }
00281           }
00282         }
00283       }
00284 
00285 #if defined(TCCR0B)
00286       if (_timer == 0)
00287       {
00288         TCCR0B = prescalarbits;
00289       }
00290       else
00291 #endif
00292 #if defined(TCCR2B)
00293       {
00294         TCCR2B = prescalarbits;
00295       }
00296 #else
00297       {
00298         // dummy place holder to make the above ifdefs work
00299       }
00300 #endif
00301     }
00302     else
00303     {
00304       // two choices for the 16 bit timers: ck/1 or ck/64
00305       ocr = F_CPU / frequency / 2 - 1;
00306 
00307       prescalarbits = 0b001;
00308       if (ocr > 0xffff)
00309       {
00310         ocr = F_CPU / frequency / 2 / 64 - 1;
00311         prescalarbits = 0b011;
00312       }
00313 
00314       if (_timer == 1)
00315       {
00316 #if defined(TCCR1B)
00317         TCCR1B = (TCCR1B & 0b11111000) | prescalarbits;
00318 #endif
00319       }
00320 #if defined(TCCR3B)
00321       else if (_timer == 3)
00322         TCCR3B = (TCCR3B & 0b11111000) | prescalarbits;
00323 #endif
00324 #if defined(TCCR4B)
00325       else if (_timer == 4)
00326         TCCR4B = (TCCR4B & 0b11111000) | prescalarbits;
00327 #endif
00328 #if defined(TCCR5B)
00329       else if (_timer == 5)
00330         TCCR5B = (TCCR5B & 0b11111000) | prescalarbits;
00331 #endif
00332 
00333     }
00334     
00335 
00336     // Calculate the toggle count
00337     if (duration > 0)
00338     {
00339       toggle_count = 2 * frequency * duration / 1000;
00340     }
00341     else
00342     {
00343       toggle_count = -1;
00344     }
00345 
00346     // Set the OCR for the given timer,
00347     // set the toggle count,
00348     // then turn on the interrupts
00349     switch (_timer)
00350     {
00351 
00352 #if defined(OCR0A) && defined(TIMSK0) && defined(OCIE0A)
00353       case 0:
00354         OCR0A = ocr;
00355         timer0_toggle_count = toggle_count;
00356         bitWrite(TIMSK0, OCIE0A, 1);
00357         break;
00358 #endif
00359 
00360       case 1:
00361 #if defined(OCR1A) && defined(TIMSK1) && defined(OCIE1A)
00362         OCR1A = ocr;
00363         timer1_toggle_count = toggle_count;
00364         bitWrite(TIMSK1, OCIE1A, 1);
00365 #elif defined(OCR1A) && defined(TIMSK) && defined(OCIE1A)
00366         // this combination is for at least the ATmega32
00367         OCR1A = ocr;
00368         timer1_toggle_count = toggle_count;
00369         bitWrite(TIMSK, OCIE1A, 1);
00370 #endif
00371         break;
00372 
00373 #if defined(OCR2A) && defined(TIMSK2) && defined(OCIE2A)
00374       case 2:
00375         OCR2A = ocr;
00376         timer2_toggle_count = toggle_count;
00377         bitWrite(TIMSK2, OCIE2A, 1);
00378         break;
00379 #endif
00380 
00381 #if defined(TIMSK3)
00382       case 3:
00383         OCR3A = ocr;
00384         timer3_toggle_count = toggle_count;
00385         bitWrite(TIMSK3, OCIE3A, 1);
00386         break;
00387 #endif

00429 #endif
