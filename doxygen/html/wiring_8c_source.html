00027 // the prescaler is set so that timer0 ticks every 64 clock cycles, and the
00028 // the overflow handler is called every 256 ticks.
00029 #define MICROSECONDS_PER_TIMER0_OVERFLOW (clockCyclesToMicroseconds(64 * 256))
00030 
00031 // the whole number of milliseconds per timer0 overflow
00032 #define MILLIS_INC (MICROSECONDS_PER_TIMER0_OVERFLOW / 1000)
00033 
00034 // the fractional number of milliseconds per timer0 overflow. we shift right
00035 // by three to fit these numbers into a byte. (for the clock speeds we care
00036 // about - 8 and 16 MHz - this doesn't lose precision.)
00037 #define FRACT_INC ((MICROSECONDS_PER_TIMER0_OVERFLOW % 1000) >> 3)
00038 #define FRACT_MAX (1000 >> 3)
00039 
00040 volatile unsigned long timer0_overflow_count = 0;
00041 volatile unsigned long timer0_millis = 0;
00042 static unsigned char timer0_fract = 0;

00044 SIGNAL(TIMER0_OVF_vect)
00045 {
00046     // copy these to local variables so they can be stored in registers
00047     // (volatile variables must be read from memory on every access)
00048     unsigned long m = timer0_millis;
00049     unsigned char f = timer0_fract;
00050 
00051     m += MILLIS_INC;
00052     f += FRACT_INC;
00053     if (f >= FRACT_MAX) {
00054         f -= FRACT_MAX;
00055         m += 1;
00056     }
00057 
00058     timer0_fract = f;
00059     timer0_millis = m;
00060     timer0_overflow_count++;
00061 }

00077 unsigned long micros() {
00078     unsigned long m;
00079     uint8_t oldSREG = SREG, t;
00080     
00081     cli();
00082     m = timer0_overflow_count;
00083 #if defined(TCNT0)
00084     t = TCNT0;
00085 #elif defined(TCNT0L)
00086     t = TCNT0L;
00087 #else
00088     #error TIMER 0 not defined
00089 #endif

00092 #ifdef TIFR0
00093     if ((TIFR0 & _BV(TOV0)) && (t < 255))
00094         m++;
00095 #else
00096     if ((TIFR & _BV(TOV0)) && (t < 255))
00097         m++;
00098 #endif
00099 
00100     SREG = oldSREG;
00101     
00102     return ((m << 8) + t) * (64 / clockCyclesPerMicrosecond());

00117 /* Delay for the given number of microseconds.  Assumes a 8 or 16 MHz clock. */
00118 void delayMicroseconds(unsigned int us)
00119 {
00120     // calling avrlib's delay_us() function with low values (e.g. 1 or
00121     // 2 microseconds) gives delays longer than desired.
00122     //delay_us(us);
00123 
00124 #if F_CPU >= 16000000L
00125     // for the 16 MHz clock on most Arduino boards
00126 
00127     // for a one-microsecond delay, simply return.  the overhead
00128     // of the function call yields a delay of approximately 1 1/8 us.
00129     if (--us == 0)
00130         return;
00131 
00132     // the following loop takes a quarter of a microsecond (4 cycles)
00133     // per iteration, so execute it four times for each microsecond of
00134     // delay requested.
00135     us <<= 2;
00136 
00137     // account for the time taken in the preceeding commands.
00138     us -= 2;
00139 #else
00140     // for the 8 MHz internal clock on the ATmega168
00141 
00142     // for a one- or two-microsecond delay, simply return.  the overhead of
00143     // the function calls takes more than two microseconds.  can't just
00144     // subtract two, since us is unsigned; we'd overflow.
00145     if (--us == 0)
00146         return;
00147     if (--us == 0)
00148         return;
00149 
00150     // the following loop takes half of a microsecond (4 cycles)
00151     // per iteration, so execute it twice for each microsecond of
00152     // delay requested.
00153     us <<= 1;
00154     
00155     // partially compensate for the time taken by the preceeding commands.
00156     // we can't subtract any more than this or we'd overflow w/ small delays.
00157     us--;
00158 #endif
00159 
00160     // busy wait
00161     __asm__ __volatile__ (
00162         "1: sbiw %0,1" "\n\t" // 2 cycles
00163         "brne 1b" : "=w" (us) : "0" (us) // 2 cycles
00164     );
00165 }
