00040 int analogRead(uint8_t pin)
00041 {
00042     uint8_t low, high;
00043 
00044 #if defined(__AVR_ATmega1280__) || defined(__AVR_ATmega2560__)
00045     if (pin >= 54) pin -= 54; // allow for channel or pin numbers
00046 #else
00047     if (pin >= 14) pin -= 14; // allow for channel or pin numbers
00048 #endif
00049 
00050 #if defined(ADCSRB) && defined(MUX5)
00051     // the MUX5 bit of ADCSRB selects whether we're reading from channels
00052     // 0 to 7 (MUX5 low) or 8 to 15 (MUX5 high).
00053     ADCSRB = (ADCSRB & ~(1 << MUX5)) | (((pin >> 3) & 0x01) << MUX5);
00054 #endif
00055   
00056     // set the analog reference (high two bits of ADMUX) and select the
00057     // channel (low 4 bits).  this also sets ADLAR (left-adjust result)
00058     // to 0 (the default).
00059 #if defined(ADMUX)
00060     ADMUX = (analog_reference << 6) | (pin & 0x07);
00061 #endif
00062 
00063     // without a delay, we seem to read from the wrong channel
00064     //delay(1);
00065 
00066 #if defined(ADCSRA) && defined(ADCL)
00067     // start the conversion
00068     sbi(ADCSRA, ADSC);
00069 
00070     // ADSC is cleared when the conversion finishes
00071     while (bit_is_set(ADCSRA, ADSC));
00072 
00073     // we have to read ADCL first; doing so locks both ADCL
00074     // and ADCH until ADCH is read.  reading ADCL second would
00075     // cause the results of each conversion to be discarded,
00076     // as ADCL and ADCH would be locked when it completed.
00077     low  = ADCL;
00078     high = ADCH;
00079 #else
00080     // we dont have an ADC, return 0
00081     low  = 0;
00082     high = 0;
00083 #endif
00084 
00085     // combine the two bytes
00086     return (high << 8) | low;
00087 }
