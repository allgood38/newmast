00057 bool TinyGPS::encode(char c)
00058 {
00059   bool valid_sentence = false;
00060 
00061   ++_encoded_characters;
00062   switch(c)
00063   {
00064   case ',': // term terminators
00065     _parity ^= c;
00066   case '\r':
00067   case '\n':
00068   case '*':
00069     if (_term_offset < sizeof(_term))
00070     {
00071       _term[_term_offset] = 0;
00072       valid_sentence = term_complete();
00073     }
00074     ++_term_number;
00075     _term_offset = 0;
00076     _is_checksum_term = c == '*';
00077     return valid_sentence;
00078 
00079   case '$': // sentence begin
00080     _term_number = _term_offset = 0;
00081     _parity = 0;
00082     _sentence_type = _GPS_SENTENCE_OTHER;
00083     _is_checksum_term = false;
00084     _gps_data_good = false;
00085     return valid_sentence;
00086   }
00087 
00088   // ordinary characters
00089   if (_term_offset < sizeof(_term) - 1)
00090     _term[_term_offset++] = c;
00091   if (!_is_checksum_term)
00092     _parity ^= c;
00093 
00094   return valid_sentence;
00095 }

00119 unsigned long TinyGPS::parse_decimal()
00120 {
00121   char *p = _term;
00122   bool isneg = *p == '-';
00123   if (isneg) ++p;
00124   unsigned long ret = 100UL * gpsatol(p);
00125   while (gpsisdigit(*p)) ++p;
00126   if (*p == '.')
00127   {
00128     if (gpsisdigit(p[1]))
00129     {
00130       ret += 10 * (p[1] - '0');
00131       if (gpsisdigit(p[2]))
00132         ret += p[2] - '0';
00133     }
00134   }
00135   return isneg ? -ret : ret;

00138 unsigned long TinyGPS::parse_degrees()
00139 {
00140   char *p;
00141   unsigned long left = gpsatol(_term);
00142   unsigned long tenk_minutes = (left % 100UL) * 10000UL;
00143   for (p=_term; gpsisdigit(*p); ++p);
00144   if (*p == '.')
00145   {
00146     unsigned long mult = 1000;
00147     while (gpsisdigit(*++p))
00148     {
00149       tenk_minutes += mult * (*p - '0');
00150       mult /= 10;
00151     }
00152   }
00153   return (left / 100) * 100000 + tenk_minutes / 6;

00158 bool TinyGPS::term_complete()
00159 {

00162     byte checksum = 16 * from_hex(_term[0]) + from_hex(_term[1]);

00167 #ifndef _GPS_NO_STATS

00172 
00173         switch(_sentence_type)
00174         {
00175         case _GPS_SENTENCE_GPRMC:
00176           _time      = _new_time;
00177           _date      = _new_date;
00178           _latitude  = _new_latitude;
00179           _longitude = _new_longitude;
00180           _speed     = _new_speed;
00181           _course    = _new_course;
00182           break;
00183         case _GPS_SENTENCE_GPGGA:
00184           _altitude  = _new_altitude;
00185           _time      = _new_time;
00186           _latitude  = _new_latitude;
00187           _longitude = _new_longitude;
00188           break;
00189         }
00190 
00191         return true;
00192       }
00193     }
00194 
00195 #ifndef _GPS_NO_STATS
00196     else
00197       ++_failed_checksum;
00198 #endif
00199     return false;
00200   }
00201 
00202   // the first term determines the sentence type
00203   if (_term_number == 0)
00204   {
00205     if (!gpsstrcmp(_term, _GPRMC_TERM))
00206       _sentence_type = _GPS_SENTENCE_GPRMC;
00207     else if (!gpsstrcmp(_term, _GPGGA_TERM))
00208       _sentence_type = _GPS_SENTENCE_GPGGA;
00209     else
00210       _sentence_type = _GPS_SENTENCE_OTHER;
00211     return false;
00212   }
00213 
00214   if (_sentence_type != _GPS_SENTENCE_OTHER && _term[0])
00215   switch((_sentence_type == _GPS_SENTENCE_GPGGA ? 200 : 100) + _term_number)
00216   {
00217     case 101: // Time in both sentences
00218     case 201:
00219       _new_time = parse_decimal();
00220       _new_time_fix = millis();
00221       break;
00222     case 102: // GPRMC validity
00223       _gps_data_good = _term[0] == 'A';
00224       break;
00225     case 103: // Latitude
00226     case 202:
00227       _new_latitude = parse_degrees();
00228       _new_position_fix = millis();
00229       break;
00230     case 104: // N/S
00231     case 203:
00232       if (_term[0] == 'S')
00233         _new_latitude = -_new_latitude;
00234       break;
00235     case 105: // Longitude
00236     case 204:
00237       _new_longitude = parse_degrees();
00238       break;
00239     case 106: // E/W
00240     case 205:
00241       if (_term[0] == 'W')
00242         _new_longitude = -_new_longitude;
00243       break;
00244     case 107: // Speed (GPRMC)
00245       _new_speed = parse_decimal();
00246       break;
00247     case 108: // Course (GPRMC)
00248       _new_course = parse_decimal();
00249       break;
00250     case 109: // Date (GPRMC)
00251       _new_date = gpsatol(_term);
00252       break;
00253     case 206: // Fix data (GPGGA)
00254       _gps_data_good = _term[0] > '0';
00255       break;
00256     case 209: // Altitude (GPGGA)
00257       _new_altitude = parse_decimal();

00259   }
00260 
00261   return false;

00280 float TinyGPS::distance_between (float lat1, float long1, float lat2, float long2) 
00281 {
00282   // returns distance in meters between two positions, both specified 
00283   // as signed decimal-degrees latitude and longitude. Uses great-circle 
00284   // distance computation for hypothetical sphere of radius 6372795 meters.
00285   // Because Earth is no exact sphere, rounding errors may be up to 0.5%.
00286   // Courtesy of Maarten Lamers
00287   float delta = radians(long1-long2);
00288   float sdlong = sin(delta);
00289   float cdlong = cos(delta);
00290   lat1 = radians(lat1);
00291   lat2 = radians(lat2);
00292   float slat1 = sin(lat1);
00293   float clat1 = cos(lat1);
00294   float slat2 = sin(lat2);
00295   float clat2 = cos(lat2);
00296   delta = (clat1 * slat2) - (slat1 * clat2 * cdlong); 
00297   delta = sq(delta); 
00298   delta += sq(clat2 * sdlong); 
00299   delta = sqrt(delta); 
00300   float denom = (slat1 * slat2) + (clat1 * clat2 * cdlong); 
00301   delta = atan2(delta, denom); 
00302   return delta * 6372795; 
00303 }
