00130 
00131     if (port == NOT_A_PIN) return;

00133     // If the pin that support PWM output, we need to turn it off

00140         uint8_t oldSREG = SREG;

00155     uint8_t bit = digitalPinToBitMask(pin);
00156     uint8_t port = digitalPinToPort(pin);
00157
