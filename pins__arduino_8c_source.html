00029 // On the Arduino board, digital pins are also used
00030 // for the analog output (software PWM).  Analog input
00031 // pins are a separate set.
00032 
00033 // ATMEL ATMEGA8 & 168 / ARDUINO
00034 //
00035 //                  +-\/-+
00036 //            PC6  1|    |28  PC5 (AI 5)
00037 //      (D 0) PD0  2|    |27  PC4 (AI 4)
00038 //      (D 1) PD1  3|    |26  PC3 (AI 3)
00039 //      (D 2) PD2  4|    |25  PC2 (AI 2)
00040 // PWM+ (D 3) PD3  5|    |24  PC1 (AI 1)
00041 //      (D 4) PD4  6|    |23  PC0 (AI 0)
00042 //            VCC  7|    |22  GND
00043 //            GND  8|    |21  AREF
00044 //            PB6  9|    |20  AVCC
00045 //            PB7 10|    |19  PB5 (D 13)
00046 // PWM+ (D 5) PD5 11|    |18  PB4 (D 12)
00047 // PWM+ (D 6) PD6 12|    |17  PB3 (D 11) PWM
00048 //      (D 7) PD7 13|    |16  PB2 (D 10) PWM
00049 //      (D 8) PB0 14|    |15  PB1 (D 9) PWM
00050 //                  +----+
00051 //
00052 // (PWM+ indicates the additional PWM pins on the ATmega168.)
00053 
00054 // ATMEL ATMEGA1280 / ARDUINO
00055 //
00056 // 0-7 PE0-PE7   works
00057 // 8-13 PB0-PB5  works
00058 // 14-21 PA0-PA7 works 
00059 // 22-29 PH0-PH7 works
00060 // 30-35 PG5-PG0 works
00061 // 36-43 PC7-PC0 works
00062 // 44-51 PJ7-PJ0 works
00063 // 52-59 PL7-PL0 works
00064 // 60-67 PD7-PD0 works
00065 // A0-A7 PF0-PF7
00066 // A8-A15 PK0-PK7
00067
